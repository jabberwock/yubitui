        let ykman_path = crate::yubikey::pin_operations::find_ykman()?;
        let mut cmd = Command::new(ykman_path);
        if let Some(s) = serial {
            cmd.args(["--device", &s.to_string()]);
        }
        cmd.args(["openpgp", "keys", "set-touch", slot, policy.as_ykman_arg(), "--force"]);

            .stderr(Stdio::piped());

            anyhow::bail!("Failed to set touch policy: {}", stderr)

    }
