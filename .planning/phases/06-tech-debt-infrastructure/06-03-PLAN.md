```rust
pub fn render_status_bar(frame: &mut Frame, area: Rect, app: &App) {
    let is_mock = app.is_mock(); // add this accessor to App

    let status_text = match app.current_screen() { /* existing */ };
    let help_text = match app.current_screen() { /* existing */ };

    let yubikey_status = if is_mock {
        if let Some(yk) = app.yubikey_state() {
            format!("Mock {} (SN: {})", yk.info.model, yk.info.serial)
        } else {
            "Mock YubiKey 5 NFC (SN: 12345678)".to_string()
        }
    } else if let Some(yk) = app.yubikey_state() {
        format!("\u{1f510} {} (SN: {})", yk.info.model, yk.info.serial)
    } else {
        "\u{274c} No YubiKey detected".to_string()
    };

    let status_line = if is_mock {
        format!("[MOCK] YubiTUI \u{2014} Hardware simulation active | {} | {}", status_text, help_text)
    } else {
        format!("{} | {} | {}", status_text, yubikey_status, help_text)
    };

    let bar_style = if is_mock {
        Style::default().bg(Color::Yellow).fg(Color::Black)
    } else {
        Style::default().bg(Color::DarkGray).fg(Color::White)
    };

    let paragraph = Paragraph::new(status_line)
        .style(bar_style)
        .block(Block::default().borders(Borders::NONE));

    frame.render_widget(paragraph, area);
}
```
