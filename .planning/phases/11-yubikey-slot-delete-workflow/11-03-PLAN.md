             _is_piv_default_mgmt_key().unwrap_or(false)
         }

         fn _is_piv_default_mgmt_key() -> anyhow::Result<bool> {
             // NOTE: Do NOT call kill_scdaemon here — the caller (detect_all) already killed it.

         IMPORTANT: The full PIV AUTHENTICATE APDU check has a startup cost problem (Pitfall 5 from research: double scdaemon kill). For v1.1, simplify: replace the three-way check with a two-way check:
         ```rust
         pub fn is_factory_default(yk: &super::YubiKeyState) -> bool {
             let no_fido2_pin = yk.fido2.as_ref().map(|f| !f.pin_is_set).unwrap_or(false);
             let zero_oath = yk.oath.as_ref().map(|o| o.credentials.is_empty()).unwrap_or(false);
             // PIV heuristic: if no slots occupied, key is likely unconfigured
             let piv_empty = yk.piv.as_ref().map(|p| p.slots.is_empty()).unwrap_or(false);
             no_fido2_pin && zero_oath && piv_empty
         }
         ```
         This avoids a separate PC/SC call entirely. The 3-way AUTHENTICATE check can be added in v2 if needed.
