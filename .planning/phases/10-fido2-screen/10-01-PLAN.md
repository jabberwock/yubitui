```rust
use anyhow::Result;

#[derive(Debug, Clone, serde::Serialize)]
pub struct Fido2State {
    pub firmware_version: Option<String>,  // formatted "5.4.3"
    pub algorithms: Vec<String>,           // e.g. ["ES256", "EdDSA"]
    pub pin_is_set: bool,                  // from options["clientPin"]
    pub pin_retry_count: u8,               // from get_pin_retries()
    pub credentials: Option<Vec<Fido2Credential>>, // None = locked (PIN not provided)
    pub supports_cred_mgmt: bool,          // from options["credMgmt" || "credentialMgmtPreview"]
}

#[derive(Debug, Clone, serde::Serialize)]
pub struct Fido2Credential {
    pub rp_id: String,          // e.g. "github.com"
    pub rp_name: Option<String>, // e.g. "GitHub" (display name of RP)
    pub user_name: String,      // user display name
    pub credential_id: Vec<u8>, // needed for delete — stored as raw bytes
}
```

3. Add public functions (all take fresh FidoKeyHid connections per call, matching OATH/PIV pattern):

- `pub fn get_fido2_device() -> Result<ctap_hid_fido2::FidoKeyHid>` — wraps FidoKeyHidFactory::create(&LibCfg::init()), maps error to anyhow. On failure, check if error string contains "access" or "permission" and include a Windows admin privilege hint in the error message (per D-12 / FIDO-07).

- `pub fn get_fido2_info() -> Result<Fido2State>` — calls get_fido2_device(), then device.get_info(), device.get_pin_retries(). Parses firmware_version from Option<u32> by extracting major=(v>>16)&0xFF, minor=(v>>8)&0xFF, patch=v&0xFF and formatting as "{major}.{minor}.{patch}". Parses pin_is_set from info.options finding "clientPin". Parses supports_cred_mgmt from info.options finding "credMgmt" or "credentialMgmtPreview". Returns Fido2State with credentials: None (info-only, no PIN provided).

- `pub fn enumerate_credentials(pin: &str) -> Result<Vec<Fido2Credential>>` — calls get_fido2_device(), then credential_management_enumerate_rps(Some(pin)), then for each Rp calls credential_management_enumerate_credentials(Some(pin), &rp.rpid_hash). Flattens into Vec<Fido2Credential>. Uses rp.public_key_credential_rp_entity.id for rp_id, rp.public_key_credential_rp_entity.name for rp_name, credential.public_key_credential_user_entity.name (with fallback to display_name then empty string) for user_name, credential.public_key_credential_descriptor.id for credential_id.

- `pub fn delete_credential(pin: &str, credential_id: &[u8]) -> Result<()>` — calls get_fido2_device(), constructs PublicKeyCredentialDescriptor with the credential_id, calls credential_management_delete_credential(Some(pin), pkcd).

- `pub fn set_pin(new_pin: &str) -> Result<()>` — calls get_fido2_device(), then device.set_new_pin(new_pin).

- `pub fn change_pin(current_pin: &str, new_pin: &str) -> Result<()>` — calls get_fido2_device(), then device.change_pin(current_pin, new_pin).
