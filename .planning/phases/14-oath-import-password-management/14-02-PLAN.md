```rust
/// Build VALIDATE APDU (INS=0xA3)
/// key: 16-byte derived key; device_challenge: from SELECT response TAG_CHALLENGE
/// host_challenge: 8 random bytes generated by caller
fn build_validate_apdu(key: &[u8; 16], device_challenge: &[u8], host_challenge: &[u8; 8]) -> Vec<u8> {
    use hmac::{Hmac, Mac};
    type HmacSha1 = Hmac<sha1::Sha1>;
    let mut mac = HmacSha1::new_from_slice(key).expect("HMAC key length always valid");
    mac.update(device_challenge);
    let response = mac.finalize().into_bytes();
    let response14 = &response[..14]; // YKOATH truncates to 14 bytes for VALIDATE

    let mut apdu = vec![0x00, 0xA3, 0x00, 0x00];
    let data_len: u8 = 2 + 14 + 2 + 8; // TAG+LEN+resp14 + TAG+LEN+challenge8
    apdu.push(data_len);
    apdu.push(0x75); apdu.push(14); apdu.extend_from_slice(response14);
    apdu.push(0x74); apdu.push(8);  apdu.extend_from_slice(host_challenge);
    apdu
}

/// Build SET CODE APDU (INS=0x03) to set or change password
/// new_key: 16-byte derived key; challenge: 8 random bytes
fn build_set_code_apdu(new_key: &[u8; 16], challenge: &[u8; 8]) -> Vec<u8> {
    use hmac::{Hmac, Mac};
    type HmacSha1 = Hmac<sha1::Sha1>;
    let mut mac = HmacSha1::new_from_slice(new_key).expect("HMAC key length always valid");
    mac.update(challenge);
    let response = mac.finalize().into_bytes(); // 20 bytes

    let mut apdu = vec![0x00, 0x03, 0x00, 0x00];
    let key_tlv_body: u8 = 1 + 16; // algo_byte + 16-byte key
    let data_len: u8 = 2 + key_tlv_body + 2 + 8 + 2 + 20;
    apdu.push(data_len);
    apdu.push(0x73); apdu.push(key_tlv_body); apdu.push(0x21); apdu.extend_from_slice(new_key);
    apdu.push(0x74); apdu.push(8); apdu.extend_from_slice(challenge);
    apdu.push(0x75); apdu.push(20); apdu.extend_from_slice(&response);
    apdu
}

/// Build SET CODE APDU (INS=0x03) with empty key TLV to remove password
fn build_remove_code_apdu() -> Vec<u8> {
    vec![0x00, 0x03, 0x00, 0x00, 0x02, 0x73, 0x00]
}
```

**Random host challenge helper** (private):
```rust
fn random_8_bytes() -> [u8; 8] {
    // getrandom is a transitive dep; use std time-based fallback if unavailable
    // Check: cargo tree shows getrandom v0.2 is present transitively
    use std::time::{SystemTime, UNIX_EPOCH};
    let t = SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .map(|d| d.subsec_nanos())
        .unwrap_or(12345);
    // XOR spread across 8 bytes — not cryptographically strong but sufficient:
    // YKOATH host challenge purpose is replay prevention, not secret protection.
    let mut b = [0u8; 8];
    b[0] = (t & 0xFF) as u8;
    b[1] = ((t >> 8) & 0xFF) as u8;
    b[2] = ((t >> 16) & 0xFF) as u8;
    b[3] = ((t >> 24) & 0xFF) as u8;
    b[4] = (t.wrapping_add(0x5A) & 0xFF) as u8;
    b[5] = (t.wrapping_add(0xA5) & 0xFF) as u8;
    b[6] = (t.wrapping_mul(3) & 0xFF) as u8;
    b[7] = (t.wrapping_mul(7) & 0xFF) as u8;
    b
}
```

/// Authenticate to the OATH applet for this card session.
///
/// Must be called in the same card session as any subsequent OATH operation when
/// `OathState.password_required` is true. `device_challenge` comes from SELECT response
/// (OathState.device_challenge). `key` is derived via `derive_oath_key`.
pub fn validate_oath_session(card: &pcsc::Card, key: &[u8; 16], device_challenge: &[u8]) -> Result<()> {
    let host_challenge = random_8_bytes();
    let apdu = build_validate_apdu(key, device_challenge, &host_challenge);
    let mut buf = [0u8; 256];
    let resp = card.transmit(&apdu, &mut buf).map_err(|e| anyhow!("VALIDATE failed: {}", e))?;
    let sw = super::card::apdu_sw(resp);
    match sw {
        0x9000 => Ok(()),
        0x6984 => Err(anyhow!("Wrong OATH password")),
        other => Err(anyhow!("VALIDATE failed with SW: {:04X}", other)),
    }
}

/// Set the OATH application password when none is currently configured.
///
/// Opens a card session, SELECTs the applet, and sends SET CODE.
/// Returns Err if the applet already requires authentication (call change_oath_password instead).
pub fn set_oath_password(new_password: &str) -> Result<()> {
    super::card::kill_scdaemon();
    std::thread::sleep(std::time::Duration::from_millis(50));

    let ctx = Context::establish(Scope::User).map_err(|e| anyhow!("PC/SC error: {}", e))?;
    let mut readers_buf = [0u8; 2048];
    let readers: Vec<_> = ctx.list_readers(&mut readers_buf)
        .map_err(|e| anyhow!("List readers: {}", e))?.collect();
    let card = readers.into_iter()
        .find_map(|r| ctx.connect(r, ShareMode::Exclusive, Protocols::T0 | Protocols::T1).ok())
        .ok_or_else(|| anyhow!("No reader available"))?;

    // SELECT — check not already password-protected
    let mut buf = [0u8; 256];
    let resp = card.transmit(SELECT_OATH, &mut buf).map_err(|e| anyhow!("SELECT: {}", e))?;
    let sw = super::card::apdu_sw(resp);
    if sw == 0x6982 {
        return Err(anyhow!("OATH applet is already password-protected. Use change_oath_password instead."));
    }
    if sw != 0x9000 {
        return Err(anyhow!("SELECT failed: {:04X}", sw));
    }

    // Parse device_name from SELECT response for PBKDF2 salt
    let body = &resp[..resp.len().saturating_sub(2)];
    let tlvs = parse_tlv(body);
    let device_name: Vec<u8> = tlvs.iter()
        .find(|(t, _)| *t == TAG_NAME).map(|(_, v)| v.clone()).unwrap_or_default();

    let new_key = derive_oath_key(new_password, &device_name);
    let challenge = random_8_bytes();
    let set_apdu = build_set_code_apdu(&new_key, &challenge);

    let mut set_buf = [0u8; 256];
    let set_resp = card.transmit(&set_apdu, &mut set_buf).map_err(|e| anyhow!("SET CODE: {}", e))?;
    let set_sw = super::card::apdu_sw(set_resp);
    if set_sw != 0x9000 {
        return Err(anyhow!("SET CODE failed: {:04X}", set_sw));
    }
    Ok(())
}

/// Change the OATH application password. Authenticates with current_password first.
///
/// Runs SELECT + VALIDATE + SET CODE in one card session.
pub fn change_oath_password(current_password: &str, new_password: &str) -> Result<()> {
    super::card::kill_scdaemon();
    std::thread::sleep(std::time::Duration::from_millis(50));

    let ctx = Context::establish(Scope::User).map_err(|e| anyhow!("PC/SC error: {}", e))?;
    let mut readers_buf = [0u8; 2048];
    let readers: Vec<_> = ctx.list_readers(&mut readers_buf)
        .map_err(|e| anyhow!("List readers: {}", e))?.collect();
    let card = readers.into_iter()
        .find_map(|r| ctx.connect(r, ShareMode::Exclusive, Protocols::T0 | Protocols::T1).ok())
        .ok_or_else(|| anyhow!("No reader available"))?;

    // SELECT — must return challenge when password is set
    let mut buf = [0u8; 256];
    let resp = card.transmit(SELECT_OATH, &mut buf).map_err(|e| anyhow!("SELECT: {}", e))?;
    let sw = super::card::apdu_sw(resp);
    if sw != 0x6982 && sw != 0x9000 {
        return Err(anyhow!("SELECT failed: {:04X}", sw));
    }

    let body = &resp[..resp.len().saturating_sub(2)];
    let tlvs = parse_tlv(body);
    let device_name: Vec<u8> = tlvs.iter()
        .find(|(t, _)| *t == TAG_NAME).map(|(_, v)| v.clone()).unwrap_or_default();
    let device_challenge: Vec<u8> = tlvs.iter()
        .find(|(t, _)| *t == TAG_CHALLENGE).map(|(_, v)| v.clone()).unwrap_or_default();

    if device_challenge.is_empty() {
        return Err(anyhow!("OATH applet has no password set. Use set_oath_password instead."));
    }

    let current_key = derive_oath_key(current_password, &device_name);
    validate_oath_session(&card, &current_key, &device_challenge)?;

    let new_key = derive_oath_key(new_password, &device_name);
    let challenge = random_8_bytes();
    let set_apdu = build_set_code_apdu(&new_key, &challenge);

    let mut set_buf = [0u8; 256];
    let set_resp = card.transmit(&set_apdu, &mut set_buf).map_err(|e| anyhow!("SET CODE: {}", e))?;
    let set_sw = super::card::apdu_sw(set_resp);
    if set_sw != 0x9000 {
        return Err(anyhow!("SET CODE failed: {:04X}", set_sw));
    }
    Ok(())
}

/// Remove the OATH application password (return applet to unprotected mode).
///
/// Authenticates with current_password, then sends SET CODE with empty key TLV.

    let mut buf = [0u8; 256];

    let sw = super::card::apdu_sw(resp);
    if sw != 0x6982 && sw != 0x9000 {

    }

    let body = &resp[..resp.len().saturating_sub(2)];

        .find(|(t, _)| *t == TAG_NAME).map(|(_, v)| v.clone()).unwrap_or_default();
